int main(int argc, char *argv[])
{

int size=9,dire=2;
int map[9][9];
getMap(map[0],size);
x=0;
y=0;
i=map[y][x];
while(i!=4){
 if(x<8){
  if(map[y][x+1]>0){
  if(dire==0){
   turnRight();
   moveForward();

 moveForward();

 turnLeft();
 moveForward();
}

 turnRight();
 moveForward();

}
}

   moveForward();

 turnRight();

 turnLeft();

 turnRight();

}

 }
